Panduan Integrasi Google Sheets

Ikuti langkah-langkah mudah di bawah ini untuk mengaktifkan database otomatis dengan Google Spreadsheet Anda.

1

Buat Spreadsheet

Buat Google Spreadsheet baru di Drive Anda, lalu beri nama sheets tersebut misal "Database BAST Online".

2

Buka Apps Script

Pilih menu Ekstensi di Spreadsheet Anda > Klik Apps Script. Salin kode skrip di bawah ini.

3

Deploy Web App

Klik Terapkan (Deploy) > Penerapan Baru > Tipe: Aplikasi Web > Akses: Siapa Saja (Anyone).

Kode Google Apps Script (`Code.gs`)

Tempel kode di bawah ini pada file default Code.gs di Google Apps Script Anda.

/* ========================================================================
GOOGLE APPS SCRIPT - DATABASE BAST ONLINE INTERAKTIF (Terintegrasi)
========================================================================
Tautan Spreadsheet Anda:
https://docs.google.com/spreadsheets/d/1-FSRDcv57_-NWZWnABIGtQ2VmvoZ4C_F390tP-eOMYE/edit
========================================================================
*/

function doPost(e) {
  // Aktifkan LockService untuk mencegah tabrakan data
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000); // Tunggu maksimal 30 detik
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      "status": "error",
      "message": "Sistem sibuk. Silakan kirim ulang beberapa saat lagi."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  try {
    var data = JSON.parse(e.postData.contents);
    
    // Membuka spreadsheet spesifik berdasarkan ID unik dari tautan Anda
    var ss = SpreadsheetApp.openById("1-FSRDcv57_-NWZWnABIGtQ2VmvoZ4C_F390tP-eOMYE");
    
    // Cari atau buat sheet bernama "DATABASE_BAST"
    var sheet = ss.getSheetByName("DATABASE_BAST");
    if (!sheet) {
      sheet = ss.insertSheet("DATABASE_BAST");
      var headers = [
        "Timestamp", "No Dokumen", "Tanggal Serah Terima", "Lokasi",
        "Nama Pihak 1", "Jabatan Pihak 1", "Perusahaan Pihak 1",
        "Nama Pihak 2", "Jabatan Pihak 2", "Perusahaan Pihak 2",
        "Daftar Barang", "Catatan Tambahan",
        "Tanda Tangan Pihak 1", "Tanda Tangan Pihak 2"
      ];
      sheet.appendRow(headers);
      
      var headerRange = sheet.getRange(1, 1, 1, headers.length);
      headerRange.setFontWeight("bold");
      headerRange.setBackground("#0f172a");
      headerRange.setFontColor("#ffffff");
      headerRange.setHorizontalAlignment("center");
      headerRange.setVerticalAlignment("middle");
      sheet.setRowHeight(1, 35);
    }
    
    var rawItems = data.items || [];
    var itemsFormatted = rawItems.map(function(item, index) {
      return (index + 1) + ". " + item.name + " (" + item.qty + " " + item.unit + ") - Kondisi: " + item.condition + (item.desc ? " [Ket: " + item.desc + "]" : "");
    }).join("\n");

    var rowData = [
      new Date(),
      data.doc_no,
      data.doc_date,
      data.doc_location,
      data.p1_name,
      data.p1_role,
      data.p1_company,
      data.p2_name,
      data.p2_role,
      data.p2_company,
      itemsFormatted,
      data.notes || "",
      "[Tanda Tangan P1]",
      "[Tanda Tangan P2]"
    ];
    
    sheet.appendRow(rowData);
    var newRowIndex = sheet.getLastRow();
    
    // Desain baris baru agar pas dengan tanda tangan gambar
    sheet.setRowHeight(newRowIndex, 75); 
    var rowRange = sheet.getRange(newRowIndex, 1, 1, rowData.length);
    rowRange.setVerticalAlignment("middle");
    rowRange.setHorizontalAlignment("left");
    
    sheet.getRange(newRowIndex, 11).setWrap(true);
    sheet.getRange(newRowIndex, 12).setWrap(true);
    
    // Sisipkan gambar tanda tangan nyata langsung ke dalam sel
    if (data.p1_signature) {
      insertBase64ImageToCell(sheet, data.p1_signature, newRowIndex, 13, "TTD_P1_" + data.doc_no.replace(/[^a-zA-Z0-9]/g, "_"));
    }
    if (data.p2_signature) {
      insertBase64ImageToCell(sheet, data.p2_signature, newRowIndex, 14, "TTD_P2_" + data.doc_no.replace(/[^a-zA-Z0-9]/g, "_"));
    }
    
    sheet.autoResizeColumns(1, 12);
    
    return ContentService.createTextOutput(JSON.stringify({
      "status": "success",
      "message": "Dokumen BAST " + data.doc_no + " berhasil diarsipkan dengan tanda tangan visual!"
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      "status": "error",
      "message": "Internal Server Error: " + error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function insertBase64ImageToCell(sheet, base64DataUri, row, col, filename) {
  try {
    if (!base64DataUri || base64DataUri.indexOf("data:image/png;base64,") !== 0) {
      return;
    }
    var rawBase64 = base64DataUri.split(",")[1];
    var decodedBytes = Utilities.base64Decode(rawBase64);
    var blob = Utilities.newBlob(decodedBytes, "image/png", filename + ".png");
    var imageObj = sheet.insertImage(blob, col, row);
    
    var originalWidth = imageObj.getWidth();
    var originalHeight = imageObj.getHeight();
    var targetHeight = 65; 
    var targetWidth = (originalWidth / originalHeight) * targetHeight;
    
    imageObj.setWidth(targetWidth);
    imageObj.setHeight(targetHeight);
    imageObj.setAnchorCellXOffset(15);
    imageObj.setAnchorCellYOffset(5);
    
  } catch (err) {
    sheet.getRange(row, col).setValue("[Gagal memuat visual TTD]");
  }
}
